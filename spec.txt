Pattern -> | '(' '?<' Name '>' Expr ')'
         | Expr
Name    -> [a-zA-Z_0-9]+
Expr -> '{' [Pair (',' Pair)*] '}'
         | '[' [['*' ','] Pattern (',' ['*' ','] Pattern)*] ']'
         | '<' [Pattern (',' Pattern)*] '>'
         | '**/' Pattern
         | '*/' Pattern
         | '_'
         | "true"
         | "false"
         | "null"
         | Number
         | String
         | Regex
         | '(' '!<' Name '>' Type ')'
Pair    -> String ':' (Pattern | '_')
         | '_' ':' Pattern
Number  -> [0-9]+
Regex   -> '#' String
String  -> '"' [a-zA-Z_ 0-9]* '"'
Type    -> string
         | number
         | boolean
         | regex
